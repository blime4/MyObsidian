```c++
class Solution {
public:
    bool canConstruct(string s, string t) {
        unordered_map<char,int> mt;
        for(char val : t) ++mt[val];
        for(char val : s){
            if(mt.count(val) && mt[val]>0) --mt[val];
            else return false;
        }
        return true;
    }
};
```
