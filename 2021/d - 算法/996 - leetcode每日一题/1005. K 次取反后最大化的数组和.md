class Solution {
public:
    int largestSumAfterKNegations(vector<int>& v, int k) {
        int neg = 0;
        int ans = 0;
        sort(v.begin(),v.end());
        for(int val : v) if(val<0) neg++;
        if(k<=neg){
            for(int i = 0;i<v.size();i++){
                if(i<k) ans -= v[i];
                else ans += v[i];
            }
            return ans;
        }
        else{
            if((k-neg)&1){
                for(int i = 0;i<neg;i++){
                    v[i] = -v[i];
                }
                sort(v.begin(),v.end());
                ans = -v[0];
                for(int i = 1;i<v.size();i++){
                    ans += v[i];
                }
                return ans;
            }
            else{
                for(int i = 0;i<v.size();i++){
                    if(i<neg) ans -= v[i];
                    else ans += v[i];
                }
                return ans;
            }
        }
    }
};
