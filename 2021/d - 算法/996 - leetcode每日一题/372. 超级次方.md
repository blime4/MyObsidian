class Solution {
public:
    const int mod = 1337;
    int fpow(int a,int n){
        int ans = 1;
        while(n){
            if(n&1) ans = (a%mod)*(ans%mod)%mod;
            a = (a%mod)*(a%mod)%mod;
            n >>= 1;
        }
        return ans%mod;

    }
    int superPow(int a, vector<int>& b) {
        int n = b.size();
        a = a%mod;
        int ans = fpow(a,b[0]);
        for(int i=1;i<n;++i){
            ans = fpow(ans,10)*fpow(a,b[i])%mod;
        }
        return ans;
    }
};
