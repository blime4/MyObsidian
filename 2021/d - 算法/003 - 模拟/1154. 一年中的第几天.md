
# 1154. 一年中的第几天

```c++

class Solution {
public:
    int dayOfYear(string date) {
        int days[12] = {31,28,31,30,31,30,31,31,30,31,30,31};
        int y = stoi(date.substr(0,4));
        int m = stoi(date.substr(5,2));
        int d = stoi(date.substr(8,2));

        int ans = 0;
        if( (y%4 == 0 && y%100 !=0 ) || (y%400 == 0)) days[1]++;

        for(int i=0;i<m-1; ++i){
            ans += days[i];
        }

        ans += d;
        return ans;
    }
};
```
