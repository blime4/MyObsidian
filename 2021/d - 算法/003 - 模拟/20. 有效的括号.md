class Solution {
public:
    bool isValid(string s) {
        stack<char> stk;
        for(auto val: s){
            if(val == '(' || val == '[' or val == '{') stk.push(val);
            else{
                if(stk.empty()) return false;
                if(val == ')') if(stk.top()=='(') stk.pop(); else return false;
                if(val == ']') if(stk.top()=='[') stk.pop(); else return false;
                if(val == '}') if(stk.top()=='{') stk.pop(); else return false;
            }

        }
        return stk.empty();
    }
};
