class Solution {
public:
int waysToStep(int n) {
        long a = 1, b = 2, c = 4, d = 7;
        if(n == 1) return 1;
        if(n == 2) return 2;
        if(n == 3) return 4;
        if(n == 4) return 7;
        for(int i = 0; i < n-4 ;i++){
            a = b;
            b = c;
            c = d;
            d = a + b + c;
            d = d % 1000000007;
        }
        return d;
    }
};
