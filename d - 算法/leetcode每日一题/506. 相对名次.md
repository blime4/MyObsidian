class Solution {
public:
    vector<string> change {"Gold Medal","Silver Medal","Bronze Medal"};
    vector<string> findRelativeRanks(vector<int>& v) {
        int n = v.size();
        vector<int>cop;
        for(int val: v) cop.push_back(val);
        sort(cop.begin(),cop.end(),greater<int>());
        unordered_map<int,int> m;
        int i = 0;
        for(int val:cop) m[val] = i++;
        vector<string> ans;
        for(int val:v){
            int index = m[val];
            if(index<3) ans.push_back(change[index]);
            else ans.push_back(to_string(index+1));
        } 
        return ans;

    }
};
