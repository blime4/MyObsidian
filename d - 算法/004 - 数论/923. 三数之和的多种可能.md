class Solution {
public:
    // 组合数的方法
    int threeSumMulti(vector<int>& v, int target) {
        constexpr int kMax = 100;
        constexpr int kMod = 1e9 + 7;
        vector<long> m(kMax+1, 0);
        for(int val : v) ++m[val];
        long ans = 0;
        for(int i = 0; i <= target; ++i){
            for(int j = i; j<= target; ++j){
                const int k = target - i - j;
                if(k<0 || k > kMax || k < j) continue;
                if(!m[i] || !m[j] || !m[k] ) continue;
                if( i== j && j == k) ans += m[i]*(m[i]-1)*(m[i]-2)/6;
                else if(i==j && j!=k) ans += m[i]*(m[i]-1)/2 * m[k];
                else if(i!=j && j==k) ans += m[i] * m[j]*(m[j]-1)/2;
                else ans += m[i]*m[j]*m[k];
            }
        }
        return ans%kMod;
    }
};
