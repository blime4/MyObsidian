[[马拉车算法]]

```c++
class Solution {
public:
    pair<int, int> huiwen(const string & s, int l , int r){
        while(l>=0 && r < s.size() && s[l] == s[r]){
            --l;
            ++r;
        }
        return {l+1,r-1};
    }
    string longestPalindrome(string s) {
        int start = 0, end = 0;
        for( int i=0; i< s.size(); ++i){
            auto [l1, r1] = huiwen(s,i,i);
            auto [l2, r2] = huiwen(s,i,i+1);

            if(r1 - l1 > end - start){
                start = l1;
                end = r1;
            }
            if(r2 - l2 > end - start){
                start = l2;
                end = r2;
            }
        }
        return s.substr(start, end - start +1);
    }
};
```
