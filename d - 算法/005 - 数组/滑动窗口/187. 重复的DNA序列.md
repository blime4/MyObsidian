```c++
class Solution {
public:
    vector<string> findRepeatedDnaSequences(string s) {
        vector<string> ans;
        unordered_map<string , int> m;
        int n = s.size();
        for(int i=0; i<= n-10; i++){
            string tmp = s.substr(i,10);
            if(++m[tmp]==2) ans.push_back(tmp);
        }
        return ans;
    }
};
```
