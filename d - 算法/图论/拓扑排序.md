vector<vector<int>> v (n+1,vector<int>(n+1,0));
vector<bool>visited(n+1,false);
vector<int>indegree(n+1,0);

vector<int> ans;

// 1. 建立邻接表
for(int i=0;i<m;i++){
	cin >> x >> y;
	v[x][y] = 1;
	indegree[y]++;
}

// 2. 将入度为零入队
multiset<int> q;
for(int i= 1;i<=n;i++){
	if(indegree[i]==0){
		q.insert(i);
		visited[i]=true;
	}
}

// 3. 遍历
while(!q.empty()){
	int cur = *q.begin();
	q.erase(q.begin());
	ans.push_back(cur);
	// 4. 相邻节点入度减减
	for(int i=0;i<v[cur].size();i++){
		if(v[cur][i]==0) continue;
		if(--indegree[i]==0 && !visited[i]
		{
			visited[i] = true;
			q.insert(i);
		}
	}
}
