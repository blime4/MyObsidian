```c++
class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        // 滑动窗口的变体
        unordered_set<int> s;
        for(int i = 0; i < nums.size(); ++i){
            if(s.count(nums[i])) return true;
            s.insert(nums[i]);
            if(s.size() > k) s.erase(nums[i-k]);
        }
        return false;
    }
};
```
