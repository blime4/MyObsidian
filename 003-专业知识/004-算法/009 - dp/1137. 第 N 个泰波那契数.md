#### [1137. 第 N 个泰波那契数](https://leetcode-cn.com/problems/n-th-tribonacci-number/)

```c++
class Solution {
public:
    int tribonacci(int n) {
        if(n < 2) return n;
        if(n == 2) return 1;
        int p1 = 0, p2 = 1, p3 = 1, ans = 0;
        for(int i = 3;i<=n;i++){
            ans = p1 + p2 + p3;
            p1 = p2;
            p2 = p3;
            p3 = ans;
        }
        return ans;
    }
};
```
