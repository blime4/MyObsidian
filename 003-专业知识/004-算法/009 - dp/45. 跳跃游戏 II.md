```c++
class Solution {
public:
    int jump(vector<int>& v) {
        int selectIdx = 0;
        int farest = 0;
        int n = v.size();
        int step = 0;
        if(n == 1) return 0;
        for(int i = 0 ; i<n-1;++i){
            farest = max(farest ,i+v[i]);
            if(selectIdx == i){
                selectIdx = farest;
                step++;
            }
        }
        return step;
    }
};
```
