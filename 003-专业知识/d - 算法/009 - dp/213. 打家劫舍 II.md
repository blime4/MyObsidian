class Solution {
public:
    int rob(vector<int>& v) {
        int n = v.size();
        if(n == 0) return 0;
        if(n == 1) return v[0];
        if(n == 2) return max(v[0],v[1]);

        return max(robRange(v,0,n-1),robRange(v,1,n));
    }

    int robRange(vector<int>& v,int start, int end){
        int first = v[start];
        int second = max(v[start],v[start+1]);
        for(int i = start + 2; i < end; i++){
            int tmp = second;
            second = max(second,first+v[i]);
            first = tmp;
        }
        return second;
    }
};
