class Solution {
public:
    ListNode* deleteDuplicates(ListNode* node) {
        if(!node) return nullptr;
        ListNode* head = node;
        while(head && head->next){
            if(head->val == head->next->val) head->next = head->next->next;
            else head = head->next;
        }
        return node;
    }
};
