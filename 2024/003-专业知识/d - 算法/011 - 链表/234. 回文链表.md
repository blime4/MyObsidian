class Solution {
public:
    bool isPalindrome(ListNode* head) {
        // 时间空间O(n)算法
        vector<int> list;
        while(head){
            list.push_back(head->val);
            head = head->next;
        }
        int l = 0, r = list.size()-1;
        while(l<r){
            if(list[l++]!=list[r--]) return false;
        }
        return true;
    }
};
