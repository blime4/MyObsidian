```c++
class Solution {
public:
    int rob(vector<int>& v) {
        int n = v.size();
        if(n == 0) return 0;
        if(n == 1) return v[0];
        vector<int> dp(n,0);
        dp[0] = v[0];
        dp[1] = max(v[0],v[1]);
        for(int i = 2;i<n;i++){
            dp[i] = max(dp[i-1],dp[i-2]+v[i]);
        }
        return dp[n-1];
    }
};
```


也可以直接用 first 和 second 来记录
```c++
class Solution {
public:
    int rob(vector<int>& v) {
        int n = v.size();
        if(n == 0) return 0;
        if(n == 1) return v[0];
        if(n == 2) return max(v[0],v[1]);

        int first = v[0];
        int second = max(v[0],v[1]);
        for(int i = 2;i<n;i++){
            int tmp = second;
            second = max(second,first+v[i]);
            first = tmp;
        }
        return second;
    }
};
```
