class Solution {
public:
    vector<int> singleNumber(vector<int>& nums) {
        unordered_map<int, int> m;
        for(int val : nums) ++m[val];
        vector<int> ans;
        for(auto [num, cnt] : m){
            if(cnt==1) ans.push_back(num);
        }
        return ans;
    }
};
