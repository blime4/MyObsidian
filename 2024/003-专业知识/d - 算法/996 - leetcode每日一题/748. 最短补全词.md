class Solution {
public:
    typedef array<int,26> ai26;
    ai26 getcnt(string& s){
        ai26 cnt{};
        for(auto val : s){
            if(isalpha(val)){
                ++cnt[tolower(val)-'a'];
            }
        }
        return cnt;
    }
    string shortestCompletingWord(string s, vector<string>& v) {
        ai26 cnt = getcnt(s);
        int idx = -1;
        bool ok = false;
        int add = 0;
        for(auto val : v){
            ai26 cnt1 = getcnt(val);
            ok = true;
            for(int i=0;i<26;i++){
                if(cnt1[i]<cnt[i]){
                    ok = false;
                    break;
                }
            }
            if(ok && (idx < 0 || val.size() < v[idx].size())){
                idx = add;
            }
            add++;
        }
        return v[idx];
    }
};
